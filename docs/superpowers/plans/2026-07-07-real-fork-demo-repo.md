**Files:**

- [ ] **Step 1: Create `scripts/gitwasm-conflict-demo.sh`**

Create `scripts/gitwasm-conflict-demo.sh` with this exact content:

```bash
#!/usr/bin/env bash
set -euo pipefail

log() {
  printf '\n==> %s\n' "$*"
}

fail() {
  printf 'gitwasm demo: %s\n' "$*" >&2
  exit 1
}

require() {
  command -v "$1" >/dev/null 2>&1 || fail "missing required command: $1"
}

repo_root="$(git rev-parse --show-toplevel 2>/dev/null)" || fail "run this script inside a git checkout"

require git
require gitwasm
require corepack

if ! command -v pnpm >/dev/null 2>&1; then
  log "pnpm is not on PATH; enabling pnpm through Corepack"
  corepack enable
fi
require pnpm

pnpm_version="$(pnpm --version)"
case "$pnpm_version" in
  10.*) ;;
  *) fail "expected pnpm 10.x, found pnpm $pnpm_version" ;;
esac

tmp="$(mktemp -d "${TMPDIR:-/tmp}/gitwasm-magic-resume-demo.XXXXXX")"
cleanup() {
  rm -rf "$tmp"
}
trap cleanup EXIT

log "creating disposable clone at $tmp/repo"
if ! git clone --quiet --local "$repo_root" "$tmp/repo" 2>/dev/null; then
  git clone --quiet "file://$repo_root" "$tmp/repo"
fi

cd "$tmp/repo"
git config user.name "gitwasm demo"
git config user.email "demo@gitwasm.dev"

log "activating committed gitwasm merge drivers"
gitwasm install
gitwasm verify

log "installing dependencies from the committed lockfile"
HUSKY=0 pnpm install --frozen-lockfile

base="$(git rev-parse HEAD)"

log "creating demo-left with left-pad@1.3.0"
git checkout --quiet -b demo-left "$base"
HUSKY=0 pnpm add -wD left-pad@1.3.0 --lockfile-only
git add package.json pnpm-lock.yaml
git commit --quiet -m "demo: add left-pad"

log "creating demo-right with is-sorted@1.0.5"
git checkout --quiet -b demo-right "$base"
HUSKY=0 pnpm add -wD is-sorted@1.0.5 --lockfile-only
git add package.json pnpm-lock.yaml
git commit --quiet -m "demo: add is-sorted"

log "merging demo-right into demo-left"
git checkout --quiet demo-left
git merge --no-edit demo-right

if git status --porcelain | grep -E '^(UU|AA|DD|AU|UA|DU|UD) ' >/dev/null; then
  git status --short
  fail "merge left unresolved conflicts"
fi

log "verifying both dependency changes survived"
grep -q 'left-pad' package.json
grep -q 'is-sorted' package.json
grep -q 'left-pad' pnpm-lock.yaml
grep -q 'is-sorted' pnpm-lock.yaml
HUSKY=0 pnpm install --frozen-lockfile

log "checking that gitwasm recorded pnpm and package.json merge verdicts"
verdict_output="$(gitwasm verdicts)"
printf '%s\n' "$verdict_output"
printf '%s\n' "$verdict_output" | grep -q 'pnpm-lock.yaml'
printf '%s\n' "$verdict_output" | grep -q 'package.json'

log "auditing recorded verdicts"
gitwasm audit

log "demo passed"
printf 'Temporary clone was removed: %s\n' "$tmp"
```

Create `.github/workflows/gitwasm-demo.yml` with this exact content:

```yaml
name: gitwasm demo

on:
  push:
    branches:
      - master
      - gitwasm-demo
  pull_request:

permissions:
  contents: read

jobs:
  app:
    name: app checks
    runs-on: ubuntu-latest
    steps:
      - name: Checkout
        uses: actions/checkout@v4

      - name: Setup Node
        uses: actions/setup-node@v4
        with:
          node-version: 22
          cache: pnpm

      - name: Setup pnpm
        run: |
          corepack enable
          corepack prepare pnpm@10.28.1 --activate

      - name: Configure self-hosted web mode
        run: |
          cat > apps/web/.env.local <<'EOF'
          NEXT_PUBLIC_APP_MODE=self-hosted
          NEXT_PUBLIC_APP_URL=http://localhost:3000
          EOF

      - name: Install dependencies
        run: HUSKY=0 pnpm install --frozen-lockfile

      - name: Run package checks
        run: |
          HUSKY=0 pnpm --filter @magic-resume/resume-schema test
          HUSKY=0 pnpm --filter @magic-resume/resume-templates test
          HUSKY=0 pnpm --filter @magic-resume/mcp test
          HUSKY=0 pnpm --filter @magic-resume/web test

      - name: Build self-hosted web app
        run: HUSKY=0 pnpm --filter @magic-resume/web build

  gitwasm:
    name: gitwasm verification
    runs-on: ubuntu-latest
    steps:
      - name: Checkout demo repo
        uses: actions/checkout@v4

      - name: Setup Node
        uses: actions/setup-node@v4
        with:
          node-version: 22
          cache: pnpm

      - name: Setup pnpm
        run: |
          corepack enable
          corepack prepare pnpm@10.28.1 --activate

      - name: Setup Rust
        uses: dtolnay/rust-toolchain@stable
        with:
          targets: wasm32-wasip1,wasm32-unknown-unknown

      - name: Build gitwasm
        run: |
          git clone --depth 1 https://github.com/gitwasm/gitwasm.git /tmp/gitwasm
          cargo build --release -p gitwasm --manifest-path /tmp/gitwasm/Cargo.toml
          echo "/tmp/gitwasm/target/release" >> "$GITHUB_PATH"

      - name: Verify committed gitwasm assets
        run: |
          gitwasm install
          gitwasm verify
          gitwasm list

      - name: Run conflict demo
        run: scripts/gitwasm-conflict-demo.sh
```
